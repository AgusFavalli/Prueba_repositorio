POSICIONARSE EN UNA CARPETA 
cd C:\Users\Agustin\Desktop\CarpetaPrueba


INICIALIZAR UN REPOSITORIO EN LA UBICACION DADA
git init


ANTES DE CONECTAR CON EL REPOSITORIO REMOTO TENEMOS QUE IDENTIFICARNOS
git config user.name "AgusFavalli" (por convencion se pone el nombre de usuario que tengamos en github)
git config user.email "agustinfavalli@..." 


COMUNICAR EL REPOSITORIO LOCAL CON EL REPOSITORIO REMOTO
git remote add origin "https://github.com/AgusFavalli/Prueba_repositorio" (Url del repositorio remoto)
                                                                          (origin: nombre de nuestro repositorio remoto, se suele usar el nombre origin por convencion)

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

VER EL ESTADO DE GIT (si aparecen archivos en rojo es que esos archivos aun no fueron agregados al repositorio local)
git status 


AGREGAR AL REPOSITORIO LOCAL DE FORMA PROVISORIA 
git add "prueba.txt" (nombre del archivo)


CONFIRMAR QUE EL ARCHIVO SE AGREGUE AL REPOSITORIO LOCAL
git commit -m "mensaje de prueba" (especificamos un mensaje para esta confirmacion que despues va a aparecer en github)

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

PASAR EL ARCHIVO AL REPOSITORIO REMOTO
git push -u origin master (origin: nombre de nuestro repositorio remoto, se suele usar el nombre origin por convencion)
                          (master: rama sobre la cual estamos posicionados)

    ----- luego de hacer esto posiblemente nos pida que validemos el usuario y contrase√±a de github--------

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

PARA CREAR UNA COPIA LOCAL DE UN REPOSITORIO REMOTO COMPLETO. (primero debemos posicionarnos en una carpeta donde queremos que se clone)
git clone https://github.com/AgusFavalli/Prueba_repositorio (Url del repositorio remoto)


PARA OBTENER LOS ULTIMOS CAMBIOS REALIZADOS POR OTROS, CUANDO YA TENEMOS UNA COPIA COMPLETA DEL REPOSITORIO EN NUESTRA MAQUINA LOCAL
git pull origin master (master: rama que queremos traer al repositorio local)

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

PARA SABER EN QUE RAMA ESTAMOS POSICIONADOS
git branch

CREAR UNA NUEVA RAMA
git branch rama1 (nombre de la nueva rama)

CAMBIAR EL NOMBRE DE UNA RAMA 
git branch -m rama1 ramita1 (primero se pone el nombre actual de la rama, luego el nuevo nombre que va a tener la rama)

PARA CAMBIAR EN QUE RAMA ESTAMOS POSICIONADOS
git checkout ramita1 (nombre de la rama a la que quiero cambiar)

ELIMINAR UNA RAMA
git branch -d ramita1 (no se puede borrar una rama sobre la cual estas posicionado, primero debes cambiarte a otra rama)

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

CREAR UN ARCHIVO (tenemos que posicionarnos en una rama y en la carpeta donde queremos crear el archivo)
touch "texto.txt"

AGREGAR EL ARCHIVO DE FORMA PROVISORIA EN LA RAMA DONDE ESTEMOS PARADOS
git add . (con "." se agregan todos los cambios)

CONFIRMAR EL ARCHIVO CREADO 
git commit -m "prueba comit archivo" (especificamos un mensaje para esta confirmacion que despues va a aparecer en github)

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

CONOCER LAS DIFERENCIAS DE CONTENIDO QUE HAY ENTRE DOS RAMAS
git diff master rama1 (en este ejemplo si en rama1 hay mas archivos la consola te va a informar que hay x candidad de nuevos archivos con sus nombres)
                      (con este mismo ejemplo si cambiamos el orden de las ramas la consola va a mostrar que hay x archivos menos o eliminados con sus nombres)

UNIFICAR UNA RAMA A OTRA (ejemplo: unificar ramita1 a master) (nos debemos posicionar en la rama que va a RECIBIR los cambios, la rama de destino)
git  merge ramita1 master (primero la rama de origen, luego la rama de destino)
